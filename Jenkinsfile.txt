pipeline {
    agent any
    stages {
        stage('Build') { 
            steps {
		
               sh  '/opt/apache-maven-3.5.2/bin/mvn -B -DskipTests clean package' 
            }
        }
		stage('Test') { 
            steps {
                sh '/opt/apache-maven-3.5.2/bin/mvn test' 
            }
            post {
                always {
                    junit 'target/surefire-reports/*.xml' 
                }
            }
        }
    }
}
